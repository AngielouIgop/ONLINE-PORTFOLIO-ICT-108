The `user` table stores important information about each user such as their full name, username, password, contact details, role, and their total current points. Every time a user throws waste into the machine, the system records it in the `wasteEntry` table. This includes the type of material (like plastic, can, or glass), how many items were thrown, how many points were earned, and the exact date and time it happened. The type of material is defined in the `materialType` table. It lists the name of each recyclable material and how many points each one is worth. This helps the system calculate how much to add to the user's total points automatically. If a user wants to redeem a reward, the `reward` table lists all the available rewards, how many points are required for each, and how many are left in stock. CREATE TABLE user (
userID INT AUTO_INCREMENT PRIMARY KEY,
fullName VARCHAR(100) NOT NULL,
userName VARCHAR(50) NOT NULL UNIQUE,
password VARCHAR(255) NOT NULL,
email VARCHAR(100),
contactNum VARCHAR(20),
zone VARCHAR(50),
position VARCHAR(50),
regDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
totalCurrentPoints INT DEFAULT 0,
role ENUM('user', 'admin') DEFAULT 'user'
);

CREATE TABLE materialType (
materialID INT AUTO_INCREMENT PRIMARY KEY,
materialName VARCHAR(50) NOT NULL,
description TEXT,
pointsPerItem INT NOT NULL
);

CREATE TABLE wasteEntry (
entryID INT AUTO_INCREMENT PRIMARY KEY,
userID INT,
materialID INT,
quantity INT NOT NULL,
pointsEarned INT NOT NULL,
dateDeposited TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
FOREIGN KEY (userID) REFERENCES user(userID),
FOREIGN KEY (materialID) REFERENCES materialType(materialID)
);

CREATE TABLE reward (
rewardID INT AUTO_INCREMENT PRIMARY KEY,
rewardName VARCHAR(100) NOT NULL,
description TEXT,
pointsRequired INT NOT NULL,
availableStock INT DEFAULT 0
);

CREATE TABLE redemption (
redemptionID INT AUTO_INCREMENT PRIMARY KEY,
userID INT,
rewardID INT,
entryID INT,
quantity INT NOT NULL,
totalPointsUsed INT NOT NULL,
redemptionDate TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
FOREIGN KEY (userID) REFERENCES user(userID),
FOREIGN KEY (rewardID) REFERENCES reward(rewardID),
FOREIGN KEY (entryID) REFERENCES wasteEntry(entryID)
);
